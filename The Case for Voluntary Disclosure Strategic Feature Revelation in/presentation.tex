\section{Motivation}
% ======================================================================

\begin{frame}{The Disclosure Dilemma}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Consider a predictive model (e.g., firm valuation):
      \vspace{6pt}
      \begin{itemize}
        \item[\textcolor{primaryTeal}{\textbullet}] \textbf{Mandatory:} All features disclosed $\Rightarrow$ baseline accuracy
        \item[\textcolor{accentCoral}{\textbullet}] \textbf{Voluntary:} Participants \textit{choose} what to reveal
      \end{itemize}
      \vspace{8pt}
      \alert{Key insight:} Participants have \textbf{private preferred outcomes} and will strategically withhold unfavorable features.
      \vspace{8pt}
      \begin{exampleblock}{Surprising Twist}
        Selective disclosure can sometimes \textbf{help} the model by filtering out noisy signals!
      \end{exampleblock}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering

\end{frame}

\begin{frame}{Why Heterogeneous Preferences Matter}
  \begin{columns}[c]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Uniform Preferences (Easy Case)}
        If \textit{all} firms want overvaluation $\Rightarrow$ assign a conservative default.\\[4pt]
        {\small Strategic withholding is trivially discouraged.}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Heterogeneous Preferences (Hard Case)}
        Firms have \textit{different} preferred directions $\Rightarrow$ no single default works.\\[4pt]
        {\small This is the realistic \& challenging setting we study.}
      \end{block}
    \end{column}
  \end{columns}

\end{frame}

\section{Problem Formulation}
% ======================================================================

\begin{frame}{Game-Theoretic Model}
  \begin{center}
  \begin{tikzpicture}[
    scale=0.88, transform shape,
    stage/.style={rectangle, draw=primaryDark, fill=primaryDark!6, rounded corners=6pt,
                  minimum width=3.2cm, minimum height=1.1cm, align=center, font=\small},
    arrow/.style={-Stealth, thick, primaryDark!80},
    label/.style={font=\scriptsize, primaryDark}
  ]
    \node[stage, fill=accentGold!15] (nature) at (0,0) {\textbf{Nature}\\draws $(u, x, y)$};
    \node[stage, fill=primaryTeal!12] (sender) at (5,0) {\textbf{Sender}\\observes $(u, x)$};
    \node[stage, fill=accentCoral!12] (channel) at (10, 0) {\textbf{Noisy Channel}\\$M \in \{\emptyset, x\}$};
    \node[stage, fill=successGreen!12] (receiver) at (5, -2.5) {\textbf{Receiver}\\predicts $\hat{y}(M)$};
    \draw[arrow] (nature) -- node[above, label] {$(u,x)$} (sender);
    \draw[arrow] (sender) -- node[above, label] {$A \in \{\emptyset, x\}$} (channel);
    \draw[arrow] (channel) |- node[right, label, pos=0.3] {$M$\; {\tiny (drops $x$ w.p.\ $q$)}} (receiver);
    \node[below=3pt of receiver, font=\scriptsize, text=accentCoral] {Sender loss: $\ell(u, \hat{y})$ \quad\textbar\quad Receiver loss: $\ell(y, \hat{y})$};
  \end{tikzpicture}
  \end{center}
  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Sender Strategy}
        $\cS = \{ A : \cY \times \cX \mapsto \{\emptyset, x\} \}$\\[2pt]
        {\small Disclose full features $x$ or withhold $\emptyset$}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Receiver Strategy}
        $\cR = \{ \hat{y} : \{\emptyset, \cX\} \mapsto \cY \}$\\[2pt]
        {\small Predict based on received message $M$}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Stackelberg Equilibrium \& Strategic Advantage}
  We model the interaction as a \textbf{screening game} (Stackelberg):
  \vspace{4pt}
  \begin{enumerate}
    \item \textcolor{primaryTeal}{\textbf{Receiver commits}} to a strategy $\hat{y} \in \cR$
    \item \textcolor{accentCoral}{\textbf{Sender best-responds:}} $\displaystyle A(u,x) = \argmin_{a \in \{\emptyset, x\}} \E[\ell(u, \hat{y}(M)) \mid A = a]$
    \item \textcolor{primaryDark}{\textbf{Outcome realized}} and losses incurred
  \end{enumerate}
  \vspace{8pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Strategic Risk (Voluntary)}
        \vspace{-6pt}
        $$R_S := \min_{\hat{y} \in \cR} \E[\ell(Y, \hat{y}(M))]$$
        \vspace{-10pt}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Vanilla Risk (Mandatory)}
        \vspace{-6pt}
        $$R_V := \E[\ell(Y, \hat{y}_V(M))]$$
        \vspace{-10pt}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{8pt}
  \begin{exampleblock}{Definition: Strategic Advantage}
    \centering A \textbf{strategic advantage} exists if $R_S < R_V$\quad\textemdash\quad voluntary disclosure \textit{strictly outperforms} mandatory.
  \end{exampleblock}
\end{frame}

\section{Regression Results}
% ======================================================================

\begin{frame}{Regression Setup}
  \begin{itemize}
    \item Outcome space: $\cY = \R$, \quad Loss: $\ell(y, \hat{y}) = (y - \hat{y})^2$ \quad (MSE)
    \item All variables $(U, X, Y)$ have \textbf{zero mean}
    \item Linear receiver: $\hat{y}(\emptyset) = 0$, \quad $\hat{y}(x) = b^\top x$
  \end{itemize}
  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Vanilla Risk (Mandatory)}
        \vspace{-4pt}
        $$R_V^\star(q) = \Var(Y) - (1{-}q)\Var(b_Y^\top X)$$
        \vspace{-10pt}
        
        {\small where $b_Y = \Var(X)^{-1}\Cov(X,Y)$ \\ is the OLS weight vector.}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Strategic Risk (Voluntary)}
        \vspace{-4pt}
        $$R_S(b,q) = \Var(Y) - (1{-}q)\E[G \cdot \Ind{F > 0}]$$
        \vspace{-10pt}
      \end{alertblock}
    \end{column}
  \end{columns}
  \vspace{8pt}
  Key scalar fields governing strategic behavior:
  \begin{center}
    \tikz{
      \node[draw=primaryTeal, fill=primaryTeal!6, rounded corners=5pt, inner sep=8pt] {
        \begin{minipage}{0.78\textwidth}
        \centering
        $F(x,u) := (2u - b^\top x)\, b^\top x$ \hfill \textcolor{primaryTeal}{\scriptsize\textbf{Sender's decision rule}}\\[4pt]
        $G(x,y) := (2y - b^\top x)\, b^\top x$ \hfill \textcolor{accentCoral}{\scriptsize\textbf{Receiver's gain function}}
        \end{minipage}
      };
    }
  \end{center}
  {\small \textbf{Sender reveals} $x$ iff $F(x,u) > 0$ \quad --- \quad disclosure benefits their preferred outcome}
\end{frame}

\begin{frame}{Sufficiency Condition for Regression}
  \begin{alertblock}{Theorem (Sufficiency)}
    A \textbf{strategic advantage} exists if $\exists\, b \in \R^d$ such that:
    $$\|U{-}Y\|_2 \;<\; \frac{\E\big[|G(X,Y)|\big] - \Var(b_Y^\top X) - \Var((b{-}b_Y)^\top X)}{4\sqrt{\Var(b_Y^\top X)}}$$
  \end{alertblock}
  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{center}
        \tikz{
          \node[draw=accentCoral, fill=accentCoral!6, rounded corners=4pt, inner sep=8pt, text width=0.92\linewidth, align=center] {
            \textbf{LHS:} $\|U - Y\|_2$\\[4pt]
            {\small Measures \textit{preference misalignment}\\between sender's preferred\\value $U$ and truth $Y$}
          };
        }
      \end{center}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
        \tikz{
          \node[draw=primaryTeal, fill=primaryTeal!6, rounded corners=4pt, inner sep=8pt, text width=0.92\linewidth, align=center] {
            \textbf{RHS:} Feature informativeness\\[4pt]
            {\small Driven by $\E[|G|]$, the expected\\ gain from observing features,\\minus sub-optimality penalty}
          };
        }
      \end{center}
    \end{column}
  \end{columns}
  \vspace{8pt}
  \begin{center}
    {\small\textit{Strategic advantage} $\iff$ \textbf{preferences are sufficiently aligned} relative to feature informativeness}
  \end{center}
\end{frame}

\begin{frame}{Necessary Condition for Regression}
  \begin{block}{Theorem (Necessity)}
    A \textbf{necessary condition} for a strategic advantage to exist:
    $$\max_b \Big( \E\big[|G(X,Y)|\big] - \Var(b_Y^\top X) - \Var((b{-}b_Y)^\top X) \Big) > 0$$
  \end{block}
  \vspace{8pt}
  \begin{exampleblock}{Key Remarks}
    \begin{itemize}
      \item The necessary condition ensures the sufficiency bound is \textbf{non-vacuous}
      \item[\textcolor{accentCoral}{$\star$}] \textbf{Both conditions are independent of channel noise $q$!}\\
            {\small $q$ controls the \textit{magnitude} of the advantage, not its \textit{existence}}
      \item If conditions hold at $b = b_Y$ (vanilla OLS weights), they hold for optimal $b$
    \end{itemize}
  \end{exampleblock}
  \vspace{6pt}
  \begin{center}
    \tikz{
      \node[draw=accentGold, fill=accentGold!10, rounded corners=5pt, inner sep=8pt, text width=0.85\textwidth, align=center] {
        \textbf{Takeaway:} Whether voluntary disclosure helps is a \textit{structural} property of the\\
        distribution\textemdash not a consequence of the channel.
      };
    }
  \end{center}
\end{frame}

\section{Gaussian Specialization}
% ======================================================================

\begin{frame}{Gaussian Specialization}
  When $(U, X, Y) \sim \mathcal{N}(0, \Sigma)$ are jointly Gaussian, the sufficiency condition admits a \textbf{closed-form} expression.
  \vspace{6pt}
  \begin{alertblock}{Theorem (Gaussian Sufficiency)}
    With receiver predictor $b = 2\Sigma_{XX}^{-1}\Sigma_{XU}$, a strategic advantage exists if:
    $$2\bigl(b_Y^\top \Sigma_{XU} - \sigma_Z^2\bigr) + \frac{8\kappa\,\sigma_Z}{s\sqrt{2\pi}}\, e^{-\alpha^2/2} \;>\; \Sigma_{XY}^\top \Sigma_{XX}^{-1} \Sigma_{XY}$$
  \end{alertblock}
  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Defined Parameters}
        \begin{itemize}
          \item $b_Y = \Sigma_{XX}^{-1}\Sigma_{XY}$ \quad {\scriptsize (OLS for $Y$)}
          \item $b_U = \Sigma_{XX}^{-1}\Sigma_{XU}$ \quad {\scriptsize (OLS for $U$)}
          \item $\sigma_Z^2 = \Sigma_{XU}^\top \Sigma_{XX}^{-1}\Sigma_{XU}$
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Conditional Parameters}
        \begin{itemize}
          \item $\kappa = \Cov(U, Y \mid X)$
          \item $s^2 = \Var(U \mid X)$
          \item $\alpha = 0$ \quad {\scriptsize (zero mean)}
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{8pt}
  \begin{center}
    \tikz{
      \node[draw=successGreen, fill=successGreen!8, rounded corners=5pt, inner sep=8pt, text width=0.85\textwidth, align=center] {
        \textbf{Why it matters:} All quantities are simple functions of the covariance matrix $\Sigma$,\\
        making the condition directly \textbf{verifiable} from data.
      };
    }
  \end{center}
\end{frame}

\section{Classification Results}
% ======================================================================

\begin{frame}{Classification Setup}
  \begin{itemize}
    \item Outcome space: $\cY = \{-1, +1\}$, \quad Loss: $\ell(y, \hat{y}) = \Ind{\hat{y} \neq y}$ \quad ($0$-$1$ loss)
    \item Preference: $U \in \{-1, +1\}$ \quad (binary preferred label)
    \item Assume WLOG: $\E[Y] > 0$ \quad (label $+1$ is the majority class)
  \end{itemize}
  \vspace{6pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Vanilla Risk}
        \vspace{-6pt}
        \begin{align*}
          R_V^\star(q) &= q\,\Prob(Y = -\sgn(\E[Y])) \\
          &\quad + (1{-}q)\,\E\!\left[\frac{1 - |\eta(X)|}{2}\right]
        \end{align*}
        \vspace{-12pt}
        
        {\small where $\eta(x) = \E[Y \mid X\!=\!x]$.}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Strategic Risk}
        For default option $i \in \{-1,+1\}$:
        \vspace{-6pt}
        \begin{align*}
          R_{\cS_i}^\star(q) &= \pi_i\, \Prob(Y\!=\!{-i} \mid U\!=\!i)\\
          &+ \pi_{-i}\Big[(1{-}q)\E\!\left[\tfrac{1-|\eta_{-i}|}{2}\right]\\
          &\quad + q\,\Prob(Y\!=\!{-i} \mid U\!=\!{-i})\Big]
        \end{align*}
        \vspace{-10pt}
      \end{alertblock}
    \end{column}
  \end{columns}
  \vspace{6pt}
  \begin{center}
    \tikz{
      \node[draw=primaryTeal, fill=primaryTeal!6, rounded corners=5pt, inner sep=8pt, text width=0.85\textwidth, align=center] {
        Only \textbf{two} possible strategic receiver strategies:
        default to $+1$ ($\cS_1$) or default to $-1$ ($\cS_{-1}$).\\
        {\small The optimal strategy depends on distribution structure and channel noise $q$.}
      };
    }
  \end{center}
\end{frame}

\begin{frame}{Classification: Stackelberg Equilibrium}
  \begin{block}{Theorem (Stackelberg Equilibrium \& Strategic Advantage)}
    Assume $\E[Y] > 0$. At max noise ($q=1$), strategy $\cS_1$ is always better. Two scenarios arise:
  \end{block}
  \vspace{4pt}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{\centering Case 1: $\cS_1$ dominates $\forall q$}
        {\small When $R_{\cS_1}^\star(0) \le R_{\cS_{-1}}^\star(0)$:}
        \vspace{-4pt}
        \begin{align*}
          &(1{-}q)\biggl[\E\!\left[\tfrac{1-|\eta|}{2}\right]\\
          &\quad - \pi_{-1}\E\!\left[\tfrac{1-|\eta_{-1}|}{2}\right]\\
          &\quad - \pi_1\Prob(Y\!=\!{-1} \mid U\!=\!1)\biggr] > 0
        \end{align*}
        \vspace{-8pt}
        
        {\scriptsize \textcolor{successGreen}{\textbf{Independent of $q$!}} \\ Analogous to regression.}
      \end{exampleblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{\centering Case 2: Strategy switch at $q^\star$}
        {\small $\exists\, q^\star \in (0,1)$. For $q \ge q^\star$: use $\cS_1$. \\For $q < q^\star$: use $\cS_{-1}$ with:}
        \vspace{-4pt}
        \begin{align*}
          &(1{-}q)\!\left[\E\!\left[\tfrac{1-|\eta|}{2}\right] - \pi_1\E\!\left[\tfrac{1-|\eta_1|}{2}\right]\right]\\[-2pt]
          &+ q\big[\Prob(Y\!=\!{-1}) - \pi_1\Prob(Y\!=\!1 \mid U\!=\!1)\big]\\[-2pt]
          &> \pi_{-1}\Prob(Y\!=\!1 \mid U\!=\!{-1})
        \end{align*}
        \vspace{-8pt}
        
        {\scriptsize \textcolor{accentCoral}{\textbf{Depends on $q$!}} \\ Noise affects existence.}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Classification: Numerical Validation}
  \begin{columns}[c]
    \begin{column}{0.50\textwidth}
      \centering
      \includegraphics[width=\linewidth]{figures/classification_strategic_advantage.png}

    \begin{column}{0.47\textwidth}

      \begin{itemize}
        \item $(Y,U) \in \{-1,+1\}^2$
        \item $P(Y\!=\!1) = 0.6$, $P(U\!=\!1) = 0.5$

      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Online Learning}
% ======================================================================

\begin{frame}{Dynamic Setting: Unknown Environment}
  \begin{alertblock}{The Challenge}
    All previous results assumed the environment (joint distribution of $(X,Y,U)$) is \textbf{known}. What if the receiver must \textbf{learn} it?
  \end{alertblock}
  \vspace{6pt}
  \textbf{Protocol:} Over $T$ rounds, at each round $t$:
  \vspace{2pt}
  \begin{center}
  \begin{tikzpicture}[
    scale=0.82, transform shape,
    step/.style={rectangle, draw=primaryDark, fill=#1, rounded corners=4pt,
                 minimum width=2.4cm, minimum height=0.7cm, align=center, font=\scriptsize\bfseries},
    arr/.style={-Stealth, thick, primaryDark!70}
  ]
    \node[step=primaryTeal!12] (s1) at (0, 0) {Select regime\\$D_t \in \{\text{M}, \text{V}\}$};
    \node[step=accentGold!15] (s2) at (3.5, 0) {Commit to\\predictor $\hat{y}_t$};
    \node[step=accentCoral!12] (s3) at (7, 0) {Sender\\best-responds};
    \node[step=successGreen!12] (s4) at (10.5, 0) {Observe $y_t$,\\update belief};
    \draw[arr] (s1) -- (s2);
    \draw[arr] (s2) -- (s3);
    \draw[arr] (s3) -- (s4);
    \draw[arr, dashed, primaryTeal] (s4.south) -- ++(0, -0.6) -| (s1.south);
  \end{tikzpicture}
  \end{center}
  \vspace{4pt}
  \textbf{Objective:} Minimize cumulative regret:
  $$\mathcal{R}_T = \sum_{t=1}^T \ell\big(y_t, \hat{y}_t(M_t)\big) - T \cdot \min\{R_V^\star, R_S^\star\}$$
\end{frame}

\begin{frame}{Bayesian Adaptive Disclosure Protocol}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{block}{Algorithm Overview}
        \begin{enumerate}
          \item \textbf{Initialize} prior $p_1(\theta)$ over parameter space $\Theta$
          \item \textbf{Each round $t$:}
          \begin{itemize}
            \item[\textcolor{primaryTeal}{\scriptsize$\blacktriangleright$}] Evaluate expected risk under belief $p_t$ for both regimes
            \item[\textcolor{accentCoral}{\scriptsize$\blacktriangleright$}] Select $(D_t, \hat{y}_t)$ minimizing expected loss
            \item[\textcolor{successGreen}{\scriptsize$\blacktriangleright$}] Observe outcome, update posterior via Bayes' rule:
          \end{itemize}
          $$p_{t+1}(\theta) \propto p_t(\theta) \cdot \Prob(M_t, y_t \mid \theta, \hat{y}_t)$$
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{Regret Guarantee}
        Under compact $\Theta$ and standard regularity:
        $$\boxed{\E[\mathcal{R}_T] \le C\sqrt{T}}$$
        \vspace{2pt}
        {\small Sublinear regret $\Rightarrow$ algorithm \textbf{converges} to the optimal disclosure regime.}
      \end{alertblock}
      \vspace{6pt}
      \begin{exampleblock}{Regret Decomposition}
        \begin{itemize}
          \item \textbf{Estimation:} $O(\sqrt{T})$ \\ {\scriptsize standard online regression}
          \item \textbf{Regime Selection:} $O(\log T)$ \\ {\scriptsize posterior concentration}
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}
